create_pitch_plot <- function(kriging_df, title_text, subtitle_text, save_path) {
  p <- ggplot(kriging_df, aes(x = x, y = y, z = var1.pred)) +
    geom_contour_filled(breaks = my_breaks) +
    scale_fill_manual(values = my_colors, name = "战术价值 Z", guide = guide_colorsteps(show.limits = TRUE)) +
    
    # 球场辅助线
    annotate("rect", xmin=0, xmax=120, ymin=0, ymax=80, fill=NA, color="black", linewidth=0.3) +
    annotate("segment", x=60, xend=60, y=0, yend=80, color="black", linewidth=0.3) +
    annotate("point", x=60, y=40, color="black", size=1.5) +
    annotate("rect", xmin=0, xmax=18, ymin=18, ymax=62, fill=NA, color="black", linewidth=0.3) +
    annotate("rect", xmin=0, xmax=6, ymin=30, ymax=50, fill=NA, color="black", linewidth=0.3) +
    annotate("point", x=12, y=40, color="black", size=1) +
    annotate("segment", x=0, xend=0, y=36, yend=44, color="black", linewidth=1.5) +
    annotate("rect", xmin=102, xmax=120, ymin=18, ymax=62, fill=NA, color="black", linewidth=0.3) +
    annotate("rect", xmin=114, xmax=120, ymin=30, ymax=50, fill=NA, color="black", linewidth=0.3) +
    annotate("point", x=108, y=40, color="black", size=1) +
    annotate("segment", x=120, xend=120, y=36, yend=44, color="black", linewidth=1.5) +
    
    coord_fixed(ratio = 1, xlim = c(-1, 121), ylim = c(-1, 81), expand = FALSE) +
    theme_void() + 
    theme(
      plot.title = element_text(size = 32, face = "bold", hjust = 0.5, margin = margin(t=5, b = 5)),
      plot.subtitle = element_text(size = 18, hjust = 0.5, color = "grey20", margin = margin(b = 10)),
      legend.title = element_text(size = 14, face = "bold"),
      legend.text = element_text(size = 12),
      legend.key.height = unit(0.8, "cm"), 
      plot.background = element_rect(fill = "white", color = NA),
      panel.background = element_rect(fill = "white", color = NA),
      plot.margin = margin(10, 10, 10, 10) 
    ) +
    labs(title = title_text, subtitle = subtitle_text)
  
  ggsave(save_path, plot = p, width = 5, height = 4, units = "in", dpi = 300, bg="white")
  return(p)
}

三、战术价值 Z 模型的计算与网格化聚合

# 计算合成最终价值 Z
def_types <- c("Goal Keeper", "Clearance", "Block", "Interception", "Ball Recovery", "Pressure", "Duel", "Shield", "Foul Committed")
error_types <- c("Miscontrol", "Dispossessed", "Dribbled Past", "Own Goal Against")

final_model_data <- full_season_cleaned %>%
  mutate(
    custom_type = case_when(
      type.name == "Pass" & pass.type.name == "Corner" ~ "Corner",
      type.name == "Pass" & pass.type.name == "Free Kick" ~ "Free Kick",
      TRUE ~ type.name
    ),
    centrality_score = 1 - (abs(location.y - 40) / 40),
    s_base = case_when(
      custom_type %in% c("Shot", "Goal Keeper") ~ 10,
      custom_type == "Corner" ~ 7, 
      custom_type == "Free Kick" ~ 6,
      type.name == "Pass" & pass.shot_assist == TRUE ~ 8,
      custom_type %in% c("Clearance", "Block") ~ 8,
      custom_type %in% c("Interception", "Ball Recovery", "Foul Won") ~ 6,
      custom_type == "Dribble" & dribble.outcome.name == "Complete" ~ 6,
      custom_type %in% c("Duel", "Pressure", "Shield") ~ 4,
      custom_type %in% c("Pass", "Carry", "Ball Receipt*") ~ 1,
      custom_type %in% error_types ~ -3,
      TRUE ~ 1
    ),
    w_space = case_when(
      custom_type == "Corner" ~ 2.5,
      custom_type == "Free Kick" ~ 2.0,
      custom_type %in% error_types ~ 0.5,
      custom_type %in% def_types ~ 1 + ((120 - location.x) / 120) + (0.5 * centrality_score),
      TRUE ~ 1 + (location.x / 120) + (0.5 * centrality_score)
    ),
    c_pressure = ifelse(under_pressure == TRUE, 1.5, 1.0),
    q_height = case_when(
      type.name == "Pass" & pass.height.name == "High Pass" ~ 0.8,
      type.name == "Pass" & pass.height.name == "Low Pass" ~ 0.9,
      TRUE ~ 1.0
    ),
    value_z = s_base * w_space * c_pressure * q_height
  ) %>% filter(!is.na(value_z))

# 空间网格聚合 (1x1米)
grid_data <- final_model_data %>%
  mutate(grid_x = round(location.x), grid_y = round(location.y)) %>%
  group_by(grid_x, grid_y) %>%
  summarise(avg_z = mean(value_z, na.rm = TRUE), count = n(), .groups = "drop")

七、球员个人实证分析：空间过滤与战术质心提取

# 封装单一球员处理流 (包含提取、未过滤出图、频次过滤出图)
process_player_kriging <- function(player_string, cn_name, position_desc) {
  # 提取基础数据
  player_base <- final_model_data %>% filter(player.name == player_string) %>% 
    mutate(grid_x = round(location.x), grid_y = round(location.y)) %>%
    group_by(grid_x, grid_y) %>% summarise(avg_z = mean(value_z, na.rm = TRUE), count = n(), .groups = "drop")
  
  # 未过滤场
  sp_raw <- player_base
  coordinates(sp_raw) <- ~grid_x + grid_y
  df_raw <- as.data.frame(krige(avg_z ~ 1, locations = sp_raw, newdata = pred_grid, model = v_model_fitted, nmax = 50))
  colnames(df_raw)[which(names(df_raw) == "grid_x")] <- "x"
  colnames(df_raw)[which(names(df_raw) == "grid_y")] <- "y"
  create_pitch_plot(df_raw, paste0(cn_name, "战术价值场 (未过滤)"), "未过滤低频异常波动点", paste0("output_figures/Player_", cn_name, "_未过滤.png"))
  
  # 频次过滤场 (过滤掉只触球过1-2次的网格，提纯常态表现)
  sp_filt <- player_base %>% filter(count >= 3)
  coordinates(sp_filt) <- ~grid_x + grid_y
  df_filt <- as.data.frame(krige(avg_z ~ 1, locations = sp_filt, newdata = pred_grid, model = v_model_fitted, nmax = 50))
  colnames(df_filt)[which(names(df_filt) == "grid_x")] <- "x"
  colnames(df_filt)[which(names(df_filt) == "grid_y")] <- "y"
  create_pitch_plot(df_filt, paste0(cn_name, "战术价值场 (频次≥3)"), position_desc, paste0("output_figures/Player_", cn_name, "_已过滤.png"))
  
  return(df_filt)
}

mkhi_df <- process_player_kriging("Henrikh Mkhitaryan", "姆希塔良", "还原前场真实的进攻走廊覆盖")

# ==========================================
# 空间特征量化表导出
# ==========================================
calculate_spatial_metrics <- function(player_df, player_name) {
  player_df %>%
    summarise(
      Player = player_name,
      Centroid_X = sum(x * ifelse(var1.pred > 0, var1.pred, 0)) / sum(ifelse(var1.pred > 0, var1.pred, 0)),
      Centroid_Y = sum(y * ifelse(var1.pred > 0, var1.pred, 0)) / sum(ifelse(var1.pred > 0, var1.pred, 0)),
      High_Threat_Area_sqm = sum(var1.pred > 8),
      Zone14_Value = sum(ifelse(x >= 80 & x <= 100 & y >= 30 & y <= 50, var1.pred, 0))
    )
}

metrics_table <- bind_rows(
  calculate_spatial_metrics(mkhi_df, "姆希塔良 (前腰)"),
  calculate_spatial_metrics(auba_df, "奥巴梅扬 (前锋)"),
  calculate_spatial_metrics(weigl_df, "魏格尔 (后腰)"),
  calculate_spatial_metrics(hummels_df, "胡梅尔斯 (中卫)")
) %>% mutate(Centroid_X = round(Centroid_X, 1), Centroid_Y = round(Centroid_Y, 1), Zone14_Value = round(Zone14_Value, 0))

write.csv(metrics_table, "output_figures/Player_核心球员空间量化对比表.csv", row.names = FALSE)
